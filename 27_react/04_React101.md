### Share state using props

```js
export default function MyApp() {
  const [count, setCount] = useState(0);

  function handleClick() {
    setCount(count + 1);
  }

  return (
    <div>
      <h1>Counters that update together</h1>
      <MyButton count={count} onClick={handleClick} />
      <MyButton count={count} onClick={handleClick} />
    </div>
  );
}


function MyButton({ count, onClick }) {
  return (
    <button onClick={onClick}>
      Clicked {count} times
    </button>
  );
}
```

## Hooks









  What are React Hooks?

  Hooks are functions that let you "hook into" React state and lifecycle features from function components.
  Introduced in React 16.8, they allow you to use state and other React features without writing a class
  component.

  Core built-in hooks:

- useState - manages local state

- useEffect - performs side effects

- useContext - consumes context

- useReducer - manages complex state logic

- useCallback - memoizes callbacks

- useMemo - memoizes values

- useRef - persists values across renders

- useImperativeHandle - customizes ref exposure

- useLayoutEffect - synchronous effect before paint

- useDebugValue - displays debug info in DevTools
  
  How Hooks Work Internally
  
  React maintains a fiber tree where each component instance has an associated fiber node. Each fiber stores a
  linked list of hooks in the order they're called.
  
  Key concepts:
1. Call Order Matters: Hooks rely on consistent call order across renders. React uses an internal cursor to
   track which hook is being called.

2. Fiber Storage: Hook state is stored on the component's fiber node in a memoizedState property as a linked
   list.

3. Dispatcher: React uses a dispatcher pattern to swap hook implementations (mounting vs updating vs error
   states).
   
   Simplified mental model:
   
   // Conceptual implementation (simplified)
   let currentFiber = null;
   let hookIndex = 0;
   
   function useState(initialState) {
   const fiber = currentFiber;
   const hooks = fiber.memoizedState || [];
   const hook = hooks[hookIndex];
   
   if (!hook) {
    // First render - initialize
    const setState = (newState) => {
      hooks[hookIndex].state = newState;
      scheduleRerender(fiber);
    };
    hooks[hookIndex] = { state: initialState, setState };
   }
   
   const currentHook = hooks[hookIndex];
   hookIndex++;
   
   return [currentHook.state, currentHook.setState];
   }
   
   Best Practices

4. Follow the Rules of Hooks
   
   // ✅ GOOD - Top level
   function MyComponent() {
   const [count, setCount] = useState(0);
   
   return <div>{count}</div>;
   }
   
   // ❌ BAD - Inside condition
   function MyComponent() {
   if (someCondition) {
    const [count, setCount] = useState(0); // Error!
   }
   }
   
   // ❌ BAD - Inside loop
   function MyComponent() {
   for (let i = 0; i < 10; i++) {
    const [count, setCount] = useState(0); // Error!
   }
   }

5. useEffect Dependencies
   
   // ✅ GOOD - Include all dependencies
   useEffect(() => {
   fetchData(userId);
   }, [userId]);
   
   // ❌ BAD - Missing dependencies
   useEffect(() => {
   fetchData(userId); // userId should be in deps
   }, []);
   
   // ✅ GOOD - Use functional updates to avoid dependencies
   const [count, setCount] = useState(0);
   useEffect(() => {
   const timer = setInterval(() => {
    setCount(c => c + 1); // No need to depend on count
   }, 1000);
   return () => clearInterval(timer);
   }, []);

6. Custom Hooks for Reusability
   
   // ✅ GOOD - Extract logic into custom hooks
   function useFetch(url) {
   const [data, setData] = useState(null);
   const [loading, setLoading] = useState(true);
   const [error, setError] = useState(null);
   
   useEffect(() => {
    let cancelled = false;
   
    fetch(url)
      .then(res => res.json())
      .then(data => {
   
        if (!cancelled) {
          setData(data);
          setLoading(false);
        }
   
      })
      .catch(err => {
   
        if (!cancelled) {
          setError(err);
          setLoading(false);
        }
   
      });
   
    return () => { cancelled = true; };
   }, [url]);
   
   return { data, loading, error };
   }
   
   // Usage
   function MyComponent() {
   const { data, loading, error } = useFetch('/api/users');
   
   if (loading) return <div>Loading...</div>;
   if (error) return <div>Error: {error.message}</div>;
   return <div>{data}</div>;
   }

7. Optimize Performance
   
   // ✅ GOOD - Memoize expensive calculations
   const expensiveValue = useMemo(() => {
   return computeExpensiveValue(a, b);
   }, [a, b]);
   
   // ✅ GOOD - Memoize callbacks to prevent re-renders
   const handleClick = useCallback(() => {
   doSomething(value);
   }, [value]);
   
   // ❌ BAD - Creating new functions on every render
   function MyComponent() {
   return (
    <ChildComponent 
      onClick={() => doSomething()} // New function every render
    />
   );
   }

8. useRef for Mutable Values
   
   // ✅ GOOD - Store mutable values that don't trigger re-renders
   function MyComponent() {
   const intervalRef = useRef(null);
   
   useEffect(() => {
    intervalRef.current = setInterval(() => {
      console.log('tick');
    }, 1000);
   
    return () => clearInterval(intervalRef.current);
   }, []);
   
   const stopTimer = () => {
    if (intervalRef.current) {
      clearInterval(intervalRef.current);
    }
   };
   
   return <button onClick={stopTimer}>Stop</button>;
   }

9. Cleanup Effects
   
   // ✅ GOOD - Always cleanup side effects
   useEffect(() => {
   const subscription = api.subscribe(userId);
   
   return () => {
    subscription.unsubscribe(); // Cleanup
   };
   }, [userId]);
   
   // ✅ GOOD - Abort fetch requests
   useEffect(() => {
   const controller = new AbortController();
   
   fetch(url, { signal: controller.signal })
    .then(res => res.json())
    .then(setData);
   
   return () => controller.abort();
   }, [url]);

10. Avoid Unnecessary State
    
    // ❌ BAD - Derived state
    function MyComponent({ items }) {
    const [count, setCount] = useState(items.length);
    
    useEffect(() => {
     setCount(items.length);
    }, [items]);
    
    return <div>{count}</div>;
    }
    
    // ✅ GOOD - Calculate during render
    function MyComponent({ items }) {
    const count = items.length; // Just derive it
    return <div>{count}</div>;
    }

11. useReducer for Complex State
    
    // ✅ GOOD - Use reducer for related state updates
    function useFormState(initialState) {
    const [state, dispatch] = useReducer(
     (state, action) => {
       switch (action.type) {
    
         case 'SET_FIELD':
           return { ...state, [action.field]: action.value };
         case 'RESET':
           return initialState;
         default:
           return state;
    
       }
     },
     initialState
    );
    
    return [state, dispatch];
    }

12. Custom Hook Naming
    
    // ✅ GOOD - Always prefix with 'use'
    function useWindowSize() {
    const [size, setSize] = useState({
     width: window.innerWidth,
     height: window.innerHeight
    });
    
    useEffect(() => {
     const handleResize = () => {
       setSize({
    
         width: window.innerWidth,
         height: window.innerHeight
    
       });
     };
    
     window.addEventListener('resize', handleResize);
     return () => window.removeEventListener('resize', handleResize);
    }, []);
    
    return size;
    }

13. Avoid Stale Closures
    
    // ❌ BAD - Stale closure
    function MyComponent() {
    const [count, setCount] = useState(0);
    
    useEffect(() => {
    const interval = setInterval(() => {
      setCount(count + 1); // count is stale!
    }, 1000);
    return () => clearInterval(interval);
    }, []); // Missing dependency
    
    return <div>{count}</div>;
    }
    
    // ✅ GOOD - Use functional update
    function MyComponent() {
    const [count, setCount] = useState(0);
    
    useEffect(() => {
    const interval = setInterval(() => {
      setCount(c => c + 1); // Always current
    }, 1000);
    return () => clearInterval(interval);
    }, []);
    
    return <div>{count}</div>;
    }
    
    Key Takeaways

14. Hooks must be called in the same order on every render

15. Only call hooks at the top level - never in loops, conditions, or nested functions

16. Only call hooks from React functions - function components or custom hooks

17. Always specify all effect dependencies or use ESLint plugin to catch issues

18. Name custom hooks starting with "use" for linting and clarity

19. Clean up side effects to prevent memory leaks

20. Use the right hook for the job - don't overcomplicate with unnecessary state
    
    The React team provides an ESLint plugin (eslint-plugin-react-hooks) that enforces these rules
    automatically.
